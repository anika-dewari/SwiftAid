## Usage

```tsx
import TopSheet from '@/components/ui/top-sheet'
```

```tsx
<TopSheet>
  <TopSheetTrigger asChild>
    <Button>Open Sheet</Button>
  </TopSheetTrigger>
  <TopSheetContent>
    <TopSheetHeader>
      <TopSheetTitle>Title</TopSheetTitle>
      <TopSheetDescription>Description</TopSheetDescription>
    </TopSheetHeader>
    content..
    <TopSheetFooter>
      <TopSheetClose asChild>
        <Button>Close</Button>
      </TopSheetClose>
    </TopSheetFooter>
  </TopSheetContent>
</TopSheet>
```
