## Usage

```tsx
import { CursorHighlight } from "@/components/ui/cursor-highlight";
```

```tsx
<CursorHighlight
    className="text-2xl sm:text-3xl md:text-4xl font-bold"
    gradient="from-pink-500 via-red-500 to-yellow-500"
    showPointer={true}
    pointerClassName="text-pink-500"
    direction="right"
    rectangle="bg-pink-200 dark:bg-green-950 border-blue dark:border-white/20 rounded-lg p-4"
```

## API Reference

### Cursor Highlight

Text is revealed on scroll with pointer, gradient, and border effects.

  rows={[

      prop: "text",

      description: "Text content to be highlighted. If not provided, `children` will be used."
    },
    {
      prop: "children",
      type: "ReactNode",
      default: "-",
      description: "Any React node to be highlighted, used when `text` is not provided."
    },
    {
      prop: "className",
      type: "string",
      default: "-",
      description: "Custom classes for styling the text."
    },
    {
      prop: "gradient",
      type: "string",
      default: `purple gradient`,
      description: "Tailwind gradient classes used for the highlight animation."
    },
    {
      prop: "direction",
      type: `"left" | "right" | "top" | "bottom"`,
      default: `"left"`,
      description: "The animation direction for the gradient filling effect."
    },
    {
      prop: "animate",
      type: "boolean",
      default: "false",
      description: "Controls whether the highlight animation plays."
    },
    {
      prop: "duration",
      type: "number",
      default: "2",
      description: "Duration (in seconds) of the highlight animation."
    },
    {
      prop: "showPointer",
      type: "boolean",
      default: "false",
      description: "Whether to display a moving pointer along the highlight."
    },
    {
      prop: "pointerDuration",
      type: "number",
      default: "1",
      description: "Duration (in seconds) of the pointer's movement."
    },
    {
      prop: "pointerClassName",
      type: "string",
      default: "-",
      description: "Custom classes for styling the pointer."
    },
    {
      prop: "containerClassName",
      type: "string",
      default: "-",
      description: "Custom classes for styling the outer container element."
    },
    {
      prop: "rectangle",
      type: "boolean | string",
      default: "false",
      description: "Enables a rectangle border around the text."
    }
  ]}
