## Usage

```tsx
import { Glass } from "@/components/ui/glass";
```

```tsx
<Glass
  width={200}
  height={200}
  borderRadius={16}
  tintOpacity={0.15}
  blur={4}
  ripple
  followMouse
  className="m-4"
  onClick={() => console.log("Glass clicked!")}
>
  <div className="flex items-center justify-center h-full w-full text-white font-bold">
    Interactive Surface
  </div>
</Glass>
```
