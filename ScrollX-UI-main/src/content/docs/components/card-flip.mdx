## Usage

```tsx
import {
  CardFlip,
  CardFlipFront,
  CardFlipBack,
  CardFlipHeader,
  CardFlipFooter,
  CardFlipTitle,
  CardFlipDescription,
  CardFlipContent,
  CardFlipAction,
} from "@/components/ui/card-flip";
```

```tsx
<CardFlip>
  <CardFlipFront>
    <CardFlipHeader>
      <CardFlipTitle>Front Title</CardFlipTitle>
    </CardFlipHeader>
    <CardFlipContent>
      <p>Front Content</p>
    </CardFlipContent>
  </CardFlipFront>

  <CardFlipBack>
    <CardFlipHeader>
      <CardFlipTitle>Back Title</CardFlipTitle>
    </CardFlipHeader>
    <CardFlipContent>
      <p>Back Content</p>
    </CardFlipContent>
  </CardFlipBack>
</CardFlip>
```
