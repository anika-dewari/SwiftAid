## Usage

```tsx
import {
  SideSheet,
  SideSheetTrigger,
  SideSheetContent,
  SideSheetHeader,
  SideSheetTitle,
  SideSheetDescription,
  SideSheetFooter,
  SideSheetClose,
} from "@/components/ui/side-sheet";
```

```tsx
<SideSheet side="right"> // or left
  <SideSheetTrigger asChild>
    <Button>Open Sheet</Button>
  </SideSheetTrigger>
  <SideSheetContent>
    <SideSheetHeader>
      <SideSheetTitle>Title</SideSheetTitle>
      <SideSheetDescription>Description</SideSheetDescription>
    </SideSheetHeader>
    content..
    <SideSheetFooter>
      <SideSheetClose asChild>
        <Button>Close</Button>
      </SideSheetClose>
    </SideSheetFooter>
  </SideSheetContent>
</SideSheet>
```
