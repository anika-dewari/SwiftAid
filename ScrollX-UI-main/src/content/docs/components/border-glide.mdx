## Usage

```tsx
import { 
  BorderGlide, 
  BorderGlideCard, 
  BorderGlideContent, 
  BorderGlideHeader, 
  BorderGlideFooter, 
  BorderGlideTitle, 
  BorderGlideDescription 
} from "@/components/ui/border-glide"
```

```tsx
<BorderGlide>
  <BorderGlideCard>
    <BorderGlideContent>
      <BorderGlideHeader>
        <BorderGlideTitle>Title One</BorderGlideTitle>
        <BorderGlideDescription>Description text</BorderGlideDescription>
      </BorderGlideHeader>
      <BorderGlideFooter>Footer content</BorderGlideFooter>
    </BorderGlideContent>
  </BorderGlideCard>
</BorderGlide>
```
