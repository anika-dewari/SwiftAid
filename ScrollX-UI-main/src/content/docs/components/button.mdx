## Usage

```tsx
import { Button } from "@/components/ui/button";
```

```tsx
<Button>Default</Button>
<Button variant="link">Link</Button>
<Button variant="success">Success</Button>
```
```tsx
<Button variant="gradient" size="lg">
  Get Started
</Button>
```
