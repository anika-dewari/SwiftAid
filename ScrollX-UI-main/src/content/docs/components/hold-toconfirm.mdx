## Usage

```tsx
import { HoldToConfirm } from "@/components/ui/hold-toconfirm";
```

```tsx
<HoldToConfirm
      variant="style"
      size="dimension"
      duration={time}
      onConfirm={callback}
      animation="border-or-fill"
      fillClassName="styles"
      confirmedChildren={content}
      confirmedClassName="styles"
      resetAfter={time}
      showProgressOnConfirm={boolean}
    >
      content
    </HoldToConfirm>
```
