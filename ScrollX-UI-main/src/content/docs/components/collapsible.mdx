## Usage

```tsx
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@/components/ui/collapsible"
```

```tsx
<Collapsible>
  <CollapsibleTrigger>What is ScrollX UI?</CollapsibleTrigger>
  <CollapsibleContent>
    ScrollX UI is a collection of modern, animated, and customizable UI
    components built for React and Next.js.
  </CollapsibleContent>
</Collapsible>
```
